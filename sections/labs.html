Usage Example 




    

// Import the VdHexGrid component
import { VdHexGrid } from './js/hex-grid.js';

// Initialize the hex grid
const grid = new VdHexGrid({
    element: document.getElementById('hex-demo-container'),
    canvas: document.getElementById('hex-demo'),
    size: 30,        // Hex size in pixels
    width: 15,       // Number of columns
    height: 10       // Number of rows
});

// Listen for hex selection
grid.on('select', (hex) => {
    console.log('Selected hex:', hex.q, hex.r);
    // hex.q = axial q coordinate
    // hex.r = axial r coordinate
});

// Update grid dynamically
grid.setSize(40);
grid.setDimensions(20, 15);
